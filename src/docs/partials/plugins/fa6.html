Font Awesome 6

You can use this plugin to set the global default icons to FA6. This plugin requires the FA6 resources to be loaded. 


//example picker
const datetimepicker1 = new tempusDominus.TempusDominus(document.getElementById('datetimepicker1'));

//to set globally
tempusDominus.extend(window.tempusDominus.plugins.fa_six);
//or
import {load, faSixIcons} from 'tempusDominus/plugins/fa-six'
tempusDominus.extend(load);

// otherwise to set icons to an individual picker
datetimepicker1.updateOptions({ display: icons: window.tempusDominus.plugins.fa_six.faSixIcons});
//or
datetimepicker1.updateOptions({ display: icons: faSixIcons});
